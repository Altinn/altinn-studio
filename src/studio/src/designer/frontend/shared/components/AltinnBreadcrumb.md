### Default

```jsx
 <AltinnBreadcrumb
  firstLink={`${window.location.origin}/`}
  firstLinkTxt='First link text'
  secondLinkTxt='Second Link Text'
 />
```

### Two links

```jsx
 <AltinnBreadcrumb
  firstLink={`${window.location.origin}/`}
  firstLinkTxt='First link text'
  secondLink={`${window.location.origin}/#altinnbreadcrumb`}
  secondLinkTxt='Second Link Text'
 />
```
